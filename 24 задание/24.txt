class Employee:

    def __init__(self, name, salary):
        self.name = name
        self.salary = salary

    def get_name(self):
        return self.name

    def get_salary(self):
        return self.salary

employees = [
Employee("Илья", 50000),
Employee("коля", 60000),
Employee("иван", 40000)
]

for employee in employees:
    print(employee.get_name())
    print(employee.get_salary())
    print()
